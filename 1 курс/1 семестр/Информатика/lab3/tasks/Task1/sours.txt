async function searchDataByQuery(val, maxItems) {
  const query = autoLayoutKeyboard(val.trim().toLowerCase());
  const query_f = val.trim().toLowerCase();

  const filteredResults = searchData.filter(item =>
    item.searchField.toLowerCase().includes(query) ||
    item.description.toLowerCase().includes(query) ||
    (item.keyWords && item.keyWords.toLowerCase().includes(query)) ||
    item.searchField.toLowerCase().includes(query_f) ||
    item.description.toLowerCase().includes(query_f) ||
    (item.keyWords && item.keyWords.toLowerCase().includes(query_f))
  );
//  console.log('searchDataByQuery')
//  console.log(filteredResults)
  try {
    const additionalResults = await fetchSearchData(query.trim(), maxItems);
    // searchData.push(...additionalResults);
//    console.log(additionalResults)
    filteredResults.forEach((item) => item.countGroup = null)
    return additionalResults.concat(filteredResults);
  } catch (error) {
    console.error('Ошибка при загрузке дополнительных данных:', error);
  }
  return filteredResults;
}

const fetchSearchData = async (query, maxItems) => {
  try {
    const additionalData = await new Promise((resolve, reject) => {
      G2.ajax({
        process: 'QS_SEARCH_QUERY',
        g_params: [query, maxItems],
        success: function(data) {
          try {
            const additionalData = JSON.parse(data);
            resolve(additionalData);
          } catch (error) {
            console.error('Ошибка при разборе данных:', error);
            reject(error);
          }
        },
        error: function(jqXHR, textStatus, errorThrown) {
          console.error('Ошибка при получении данных:', textStatus, errorThrown);
          reject(errorThrown);
        }
      });
    });

    const uniqueAdditionalData = [];

    additionalData.forEach(item => {
      const newItem = {
        label: item.label.replace(/<(.|\n)*?>/g, '').replace(/<(.|\n)*?>/g, '').trim(),
        mainApp: item.mainApp.trim(),
        searchField: item.searchField.replace(/<(.|\n)*?>/g, '').replace(/<(.|\n)*?>/g, '').trim(),
        //        description: item.description === null ? "" : item.description.trim(),
        description: [item.description === null ? "" : item.description.trim(), item.contacts ? item.contacts.trim() : ""].filter(e => e != "&nbsp;" && e).join('<br>'),
        keyWords: item.keyWords ? item.keyWords.trim() : '',
        icon: item.icon,
        icon2: item.icon2,
        color: item.color,
        link: item.link,
        resultGroup: item.resultGroup,
        countGroup: item.countGroup,
        contacts: item.contacts ? item.contacts.trim() : ""
      };

      const isDuplicate = searchData.some(existingItem =>
        existingItem.link === newItem.link
      );

      if (!isDuplicate) {
        uniqueAdditionalData.push(newItem);
      } else {
        const existingItem = searchData.find(existingItem => existingItem.link === newItem.link);
        if (existingItem) {
          existingItem.resultGroup = newItem.resultGroup;
          existingItem.countGroup = newItem.countGroup;
        }
      }
    });

    return uniqueAdditionalData;
  } catch (error) {
    console.error('Ошибка при загрузке дополнительных данных:', error);
    throw error;
  }
};

function groupByResultGroup(items, previewItemsCount) {
  const grouped = new Map();
  items.forEach(item => {
    const group = item.resultGroup?.trim();
    const key = group ? group : groupKeyNone;

    if (!grouped.has(key)) {
      grouped.set(key, []);
    }
    grouped.get(key).push(item);
  });
  const apps = grouped.get('Центр приложений');
  let services = grouped.get('Сервисы');
  if (apps && services) {
//    console.log('Слияние групп')
    const links = new Set();
    apps.forEach((i) => {
      links.add(i.link);
      const index = i.link.indexOf('apex/f')
      if (index > 0) {
        links.add(i.link.substring(index + 5));
      }
    });
//    console.log(links)
    services = services.filter((i) => {
      const index = i.link.indexOf('apex/f')
      if (index > 0) {
        return !links.has(i.link) && !links.has(i.link.substring(index + 5));
      }
      return !links.has(i.link)
    });
//    console.log(apps)
    services.push(...apps)
    if (services[0].countGroup) {
      services[0].countGroup = apps[0].countGroup ?? apps.length;
    }
    //    services.push(...apps)
    grouped.set('Сервисы', services)
    grouped.delete('Центр приложений')
  } else if (apps) {
    grouped.set('Сервисы', apps)
    grouped.delete('Центр приложений')
  }

  const orderedGroupKeys = Array.from(grouped.keys()).sort(function(a, b) {
    const o1 = groupConfigs.get(a)?.orderNum ?? noOrderNum;
    const o2 = groupConfigs.get(b)?.orderNum ?? noOrderNum;
    return o1 - o2;
  });


  let resItemsCount = 0;
  const groups = orderedGroupKeys.map((key) => {
    const el = {
      groupName: key === groupKeyNone ? "" : key,
      groupConfig: groupConfigs.get(key === groupKeyNone ? "" : key) ??
        (() => {
          const c = Object.assign({}, groupConfigs.get(groupKeyDefault));
          c.groupKey = (' ' + key).slice(1).replaceAll(" ", "-");
          return c;
        })(),
      previewItems: grouped.get(key).slice(0, previewItemsCount),
      previewResItems: grouped.get(key).slice(0, maxPrevieResItemsCount),
      resItems: grouped.get(key).slice(0, maxResItemsCount),
      allItemsCount: grouped.get(key)[0].countGroup ?? grouped.get(key).length
    }
    resItemsCount += el.resItems.length;
    return el;
  });


  return {
    totalItems: items.length,
    totalResItems: resItemsCount,
    groups: groups
  }
}

// Keyboard Layout

function autoLayoutKeyboard(str) {
    const replacer = {
        "q": "й", "w": "ц", "e": "у", "r": "к", "t": "е", "y": "н", "u": "г",
        "i": "ш", "o": "щ", "p": "з", "[": "х", "]": "ъ", "a": "ф", "s": "ы",
        "d": "в", "f": "а", "g": "п", "h": "р", "j": "о", "k": "л", "l": "д",
        ";": "ж", "'": "э", "z": "я", "x": "ч", "c": "с", "v": "м", "b": "и",
        "n": "т", "m": "ь", ",": "б", ".": "ю", "/": "."
    };
    return str.replace(/[A-z/,.;'[\]]/g, function (x) {
        return x === x.toLowerCase() ? replacer[x] : replacer[x.toLowerCase()].toUpperCase();
    });
}

function extractCorrectFullName(possibleNames) {
  const nameParts = possibleNames.split(';');

  for (let i = 0; i < nameParts.length; i++) {
    const fullName = nameParts[i].trim();
    const words = fullName.split(' ');

    if (words.length === 4) {
      return capitalizeFullName(fullName);
    } else if (words.length === 2) {
      return capitalizeFullName(fullName);
    }
  }

//  return capitalizeFullName(nameParts[0]);
  return nameParts[0];
}

function capitalizeFirstLetter(string) {
  return string.charAt(0).toUpperCase() + string.slice(1);
}

function capitalizeFullName(fullName) {
  return fullName.split(' ')
    .map(name => name.charAt(0).toUpperCase() + name.slice(1).toLowerCase())
    .join(' ');
}

function rgbToHex(rgb) {
  // Extract R, G, B values from "rgb(R, G, B)"
  const parts = rgb.match(/\d+/g);
  if (!parts || parts.length !== 3) {
    return null; // Invalid RGB format
  }
  const r = parseInt(parts[0]);
  const g = parseInt(parts[1]);
  const b = parseInt(parts[2]);

  // Convert to hex and pad with leading zero if needed
  const toHex = (c) => ("0" + c.toString(16)).slice(-2);

  return `#${toHex(r)}${toHex(g)}${toHex(b)}`;
}
